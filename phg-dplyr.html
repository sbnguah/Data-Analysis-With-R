Creating function to configrure tables

tbl_style <- function(df){
    df %>% 
        gt::gt() %>% 
        gt::tab_options(
            table.font.size = 14, 
            table.font.names = "serif", 
            data_row.padding = gt::px(2)
        ) %>% 
        gt::opt_stylize(style = 5)
}
